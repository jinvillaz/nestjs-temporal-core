    WorkflowRun: () => MethodDecorator = ...

    Marks the main entry point method of a workflow. This decorator identifies which method serves as the workflow's main execution function. A workflow class must have exactly one method decorated with @WorkflowRun.

    Type declaration

      • (): MethodDecorator

      • Returns MethodDecorator

    Example

    @Workflow({ name: 'user-registration' })
    export class UserRegistrationWorkflow {
      @WorkflowRun()
      async execute(userData: UserData): Promise<RegistrationResult> {
        // Main workflow logic
        return { userId: '123', status: 'registered' };
      }

      @SignalMethod('cancel')
      async handleCancel(): Promise<void> {
        // Signal handling logic
      }
    }
